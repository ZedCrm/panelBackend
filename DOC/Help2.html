راهنمای گام‌به‌گام توسعه کنترلر جدید
1. تعریف موجودیت در Domain ▼

کلاس موجودیت رو در پوشه‌ی Domain\Objects بساز و از BaseDomain ارث‌بری کن.

public class Warehouse : BaseDomain
{
    public string Name { get; set; }
    public string Location { get; set; }
}
2. تعریف DTO ها در App.Contracts ▼

DTO های View، Create، Update و اینترفیس سرویس رو تعریف کن.

public class WarehouseView
{
    public int Id { get; set; }
    public string Name { get; set; }
}

public interface IWarehouseApp
{
    Task<ApiResult<List<WarehouseView>>> GetAll(Pagination pagination);
}
3. تنظیم مپینگ در ConfApp\Mapping ▼ 
public class WarehouseMapping : IEntityTypeConfiguration<Warehouse>
{
    public void Configure(EntityTypeBuilder<Warehouse> builder)
    {
        builder.ToTable("Warehouses");
        builder.HasKey(x => x.Id);
        builder.Property(x => x.Name).IsRequired().HasMaxLength(100);
    }
}
4. ساخت ریپازیتوری در ConfApp\Rep ▼ 
public interface IWarehouseRep : IBaseRep<Warehouse, int> { }

public class WarehouseRep : BaseRep<Warehouse, int>, IWarehouseRep
{
    public WarehouseRep(MyContext context) : base(context) { }
}
5. ساخت سرویس در App\Object ▼ 
public class WarehouseApp : CrudService<Warehouse, WarehouseView, WarehouseCreate, WarehouseUpdate, int>, IWarehouseApp
{
    public WarehouseApp(IWarehouseRep rep, IMapper mapper) : base(rep, mapper) { }
}
6. افزودن مپینگ در ClassMapping ▼ 
CreateMap<Warehouse, WarehouseView>();
CreateMap<WarehouseCreate, Warehouse>();
CreateMap<WarehouseUpdate, Warehouse>();
7. ثبت در DI (CRMBootstraper) ▼ 
services.AddScoped<IWarehouseRep, WarehouseRep>();
services.AddScoped<IWarehouseApp, WarehouseApp>();
8. ساخت کنترلر در API\Controllers ▼ 
[ApiController]
[Route("api/[controller]")]
public class WarehouseController : BaseController
{
    private readonly IWarehouseApp _warehouseApp;

    public WarehouseController(IWarehouseApp warehouseApp)
    {
        _warehouseApp = warehouseApp;
    }

    [HttpPost("GetAll")]
    public async Task<ActionResult> GetAll([FromBody] Pagination pagination)
    {
        return Ok(await _warehouseApp.GetAll(pagination));
    }
}
9. افزودن DbSet در MyContext ▼ 
public DbSet<Warehouse> Warehouses { get; set; }
10. اجرای میگریشن ▼ 
Add-Migration AddWarehouse
Update-Database
طراحی شده برای توسعه‌دهندگانی که می‌خوان سریع، تمیز و حرفه‌ای کد بزنن.
